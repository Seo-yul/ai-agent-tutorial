Chapter 5. Resources 심화

MCP 리소스로 LLM에게 데이터를 제공하는 방법을 배운다

5.1 Resources 개요

MCP의 Resources는 LLM이 읽을 수 있는 데이터를 서버가 외부에 노출하는 메커니즘이다. Tools가 동작(action)을 수행하는 것과 달리, Resources는 부수 효과(side effect) 없이 데이터를 제공하는 읽기 전용 인터페이스다.

Resources는 고유한 URI(Uniform Resource Identifier)로 식별되며, 클라이언트는 이 URI를 통해 데이터에 접근한다. 설정 파일, 데이터베이스 레코드, API 응답 등 다양한 형태의 데이터를 리소스로 제공할 수 있다.

┌─────────────┐ URI 요청 ┌─────────────┐ │ │ ──────────────────────▶ │ │ │ MCP 클라이언트 │ │ MCP 서버 │ │ (LLM Host) │ ◀────────────────────── │ (Resources) │ │ │ 데이터 응답 (읽기 전용) │ │ └─────────────┘ └─────────────┘
INFO

Resources는 읽기 전용이다. 데이터를 수정하거나 외부 시스템에 영향을 주는 작업은 Tools를 사용해야 한다.

5.2 정적 리소스

정적 리소스는 고정된 URI로 접근하는 리소스다. URI에 변수 부분이 없으며, 항상 동일한 엔드포인트를 통해 데이터를 반환한다. @mcp.resource() 데코레이터에 고정 URI를 전달하여 정의한다.

애플리케이션 설정 리소스

import json
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("config-server")

@mcp.resource("config://app")
def get_app_config() -> str:
    """애플리케이션 설정을 반환한다"""
    return json.dumps({
        "version": "1.0.0",
        "debug": False,
        "database": "postgresql://localhost/mydb"
    }, indent=2)

위 예제에서 config://app이라는 URI로 접근하면 애플리케이션 설정 JSON을 반환한다. 클라이언트는 이 URI를 사용하여 언제든 설정 정보를 조회할 수 있다.

시스템 상태 리소스

import platform
import psutil  # pip install psutil

@mcp.resource("system://status")
def get_system_status() -> str:
    """현재 시스템 상태를 반환한다"""
    return json.dumps({
        "os": platform.system(),
        "python_version": platform.python_version(),
        "cpu_percent": psutil.cpu_percent(),
        "memory_percent": psutil.virtual_memory().percent,
    }, indent=2)
TIP

정적 리소스라 해서 반환값이 항상 같아야 하는 것은 아니다. URI가 고정이라는 의미이며, 함수 실행 시점의 최신 데이터를 반환할 수 있다.

5.3 동적 리소스 (Resource Template)

동적 리소스는 URI 템플릿을 사용하여 파라미터를 받는 리소스다. URI 내에 {parameter} 형태의 변수를 포함하며, 클라이언트가 요청할 때 실제 값으로 치환된다. MCP 프로토콜에서는 이를 Resource Template이라 부른다.

사용자 프로필 리소스

@mcp.resource("users://{user_id}/profile")
def get_user_profile(user_id: str) -> str:
    """사용자 프로필을 반환한다"""
    profiles = {
        "1": {"name": "김철수", "role": "개발자"},
        "2": {"name": "이영희", "role": "디자이너"},
    }
    return json.dumps(profiles.get(user_id, {}), ensure_ascii=False)

users://1/profile로 요청하면 김철수의 프로필을, users://2/profile로 요청하면 이영희의 프로필을 반환한다. URI 템플릿의 {user_id} 부분이 함수의 user_id 파라미터로 자동 매핑된다.

파일 리소스

from pathlib import Path

@mcp.resource("file://project/{filepath}")
def get_project_file(filepath: str) -> str:
    """프로젝트 파일 내용을 반환한다"""
    base_dir = Path("/home/user/project")
    target = (base_dir / filepath).resolve()

    # 보안: 기본 디렉토리 외부 접근 차단
    if not str(target).startswith(str(base_dir)):
        return "접근이 거부되었다."

    if target.exists() and target.is_file():
        return target.read_text(encoding="utf-8")
    return f"파일을 찾을 수 없다: {filepath}"
WARNING

파일 시스템에 접근하는 리소스를 만들 때는 반드시 경로 탐색 공격(path traversal)을 방지하는 검증 로직을 포함해야 한다. resolve()로 정규화한 후 허용된 디렉토리 내에 있는지 확인한다.

5.4 다양한 반환 타입

리소스 함수는 다양한 타입의 데이터를 반환할 수 있다. 텍스트, JSON, 바이너리 데이터 등 상황에 맞는 형식을 선택한다.

텍스트 반환

@mcp.resource("docs://readme")
def get_readme() -> str:
    """README 문서를 반환한다"""
    return """# 프로젝트 소개
이 프로젝트는 MCP 서버 예제이다.

## 설치 방법
pip install -r requirements.txt

## 실행 방법
python server.py
"""

JSON 반환

@mcp.resource("data://products")
def get_products() -> str:
    """상품 목록을 JSON으로 반환한다"""
    products = [
        {"id": 1, "name": "노트북", "price": 1200000},
        {"id": 2, "name": "마우스", "price": 35000},
        {"id": 3, "name": "키보드", "price": 89000},
    ]
    return json.dumps(products, ensure_ascii=False, indent=2)

바이너리 데이터 반환

import base64

bytes를 반환하면 MCP SDK가 자동으로 base64 인코딩하여 전송한다. MIME 타입은 파일 확장자에서 자동 추론된다.

5.5 실전 예제: 설정 관리 서버

환경별(development, staging, production) 설정 파일을 리소스로 제공하는 MCP 서버를 만들어 보자. 각 환경에 맞는 데이터베이스, API 키, 기능 플래그 등의 설정을 조회할 수 있다. 

import json
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("config-manager")

# 환경별 설정 데이터
CONFIGS = {
    "development": {
        "env": "development",
        "debug": True,
        "database": {
            "host": "localhost",
            "port": 5432,
            "name": "myapp_dev",
            "pool_size": 5,
        },
        "cache": {
            "backend": "memory",
            "ttl": 60,
        },
        "features": {
            "new_dashboard": True,
            "beta_api": True,
            "rate_limiting": False,
        },
    },
    "staging": {
        "env": "staging",
        "debug": False,
        "database": {
            "host": "staging-db.internal",
            "port": 5432,
            "name": "myapp_staging",
            "pool_size": 10,
        },
        "cache": {
            "backend": "redis",
            "host": "staging-redis.internal",
            "ttl": 300,
        },
        "features": {
            "new_dashboard": True,
            "beta_api": True,
            "rate_limiting": True,
        },
    },
    "production": {
        "env": "production",
        "debug": False,
        "database": {
            "host": "prod-db.internal",
            "port": 5432,
            "name": "myapp_prod",
            "pool_size": 50,
        },
        "cache": {
            "backend": "redis",
            "host": "prod-redis.internal",
            "ttl": 3600,
        },
        "features": {
            "new_dashboard": False,
            "beta_api": False,
            "rate_limiting": True,
        },
    },
}


# 정적 리소스: 전체 환경 목록
@mcp.resource("config://environments")
def list_environments() -> str:
    """사용 가능한 환경 목록을 반환한다"""
    envs = [
        {"name": name, "description": f"{name} 환경 설정"}
        for name in CONFIGS
    ]
    return json.dumps(envs, ensure_ascii=False, indent=2)


# 동적 리소스: 환경별 전체 설정
@mcp.resource("config://{env}")
def get_environment_config(env: str) -> str:
    """특정 환경의 전체 설정을 반환한다"""
    config = CONFIGS.get(env)
    if config is None:
        return json.dumps(
            {"error": f"알 수 없는 환경: {env}"},
            ensure_ascii=False,
        )
    return json.dumps(config, ensure_ascii=False, indent=2)


# 동적 리소스: 환경별 특정 섹션 설정
@mcp.resource("config://{env}/{section}")
def get_config_section(env: str, section: str) -> str:
    """특정 환경의 특정 섹션 설정을 반환한다"""
    config = CONFIGS.get(env, {})
    value = config.get(section)
    if value is None:
        return json.dumps(
            {"error": f"'{env}' 환경에 '{section}' 섹션이 없다"},
            ensure_ascii=False,
        )
    return json.dumps(value, ensure_ascii=False, indent=2)


# 정적 리소스: 환경 비교
@mcp.resource("config://compare/all")
def compare_all_configs() -> str:
    """모든 환경의 주요 설정을 비교한다"""
    comparison = {}
    for env_name, config in CONFIGS.items():
        comparison[env_name] = {
            "debug": config["debug"],
            "db_host": config["database"]["host"],
            "db_pool": config["database"]["pool_size"],
            "cache_backend": config["cache"]["backend"],
            "features_enabled": sum(
                1 for v in config["features"].values() if v
            ),
        }
    return json.dumps(comparison, ensure_ascii=False, indent=2)


if __name__ == "__main__":
    mcp.run()

# 환경 목록 조회
config://environments

# production 전체 설정 조회
config://production

# staging의 database 설정만 조회
config://staging/database

# 모든 환경 비교
config://compare/all
TIP

실제 프로젝트에서는 CONFIGS 딕셔너리 대신 YAML 파일이나 환경 변수, 외부 설정 서비스(Vault, AWS Parameter Store 등)에서 설정을 불러오도록 구현한다.
