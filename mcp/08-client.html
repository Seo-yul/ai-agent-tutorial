Chapter 8. MCP 클라이언트

Python으로 MCP 클라이언트를 구현하여 서버와 통신한다

8.1 클라이언트 개요

MCP 클라이언트는 MCP 서버에 연결하여 도구 호출, 리소스 읽기, 프롬프트 가져오기를 수행하는 프로그램이다. Claude Desktop 같은 LLM 호스트가 대표적인 MCP 클라이언트이지만, Python SDK를 사용하면 직접 클라이언트를 구현할 수 있다.

직접 클라이언트를 구현하면 다음과 같은 활용이 가능하다.

  • 자동화 스크립트 — MCP 서버의 도구를 프로그래밍 방식으로 호출
  • 테스트 — MCP 서버의 동작을 자동으로 검증
  • 커스텀 LLM 호스트 — 자체 AI 애플리케이션에 MCP 서버 연동
  • CLI 도구 — 명령줄에서 MCP 서버와 상호작용
sequenceDiagram
    participant C as Client
    participant S as Server

    Note over C,S: 1. 연결 설정 (stdio 또는 Streamable HTTP)
    C->>S: 2. initialize (capabilities 교환)
    S-->>C: 지원 기능 응답

    Note over C,S: 3. 기능 사용
    C->>S: list_tools() - 도구 목록 조회
    C->>S: call_tool() - 도구 호출
    C->>S: list_resources() - 리소스 목록 조회
    C->>S: read_resource() - 리소스 읽기
    C->>S: list_prompts() - 프롬프트 목록 조회
    C->>S: get_prompt() - 프롬프트 가져오기

    Note over C,S: 4. 연결 종료
    C->>S: 세션 종료

8.2 stdio 클라이언트

stdio 클라이언트는 로컬에서 실행 중인 MCP 서버 프로세스에 stdin/stdout 파이프로 연결한다. 서버 스크립트를 직접 실행하므로 별도의 서버 프로세스를 미리 띄울 필요가 없다.

기본 연결 및 도구 호출

import asyncio
from mcp.client.session import ClientSession
from mcp.client.stdio import StdioServerParameters, stdio_client

async def main():
    # 서버 실행 파라미터 설정
    server_params = StdioServerParameters(
        command="uv",
        args=["run", "server.py"]
    )

    # 서버에 연결
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            # 초기화 (필수)
            await session.initialize()

            # 도구 목록 조회
            tools = await session.list_tools()
            print(f"사용 가능한 도구: {[t.name for t in tools.tools]}")

            # 도구 호출
            result = await session.call_tool("add", {"a": 5, "b": 3})
            print(f"결과: {result.content[0].text}")

asyncio.run(main())
WARNING

session.initialize()는 반드시 다른 메서드를 호출하기 전에 실행해야 한다. 이 과정에서 클라이언트와 서버가 프로토콜 버전과 지원 기능(capabilities)을 교환한다.

StdioServerParameters 옵션

from mcp.client.stdio import StdioServerParameters

# 다양한 실행 방식
params_uv = StdioServerParameters(
    command="uv",
    args=["run", "server.py"],
)

params_python = StdioServerParameters(
    command="python",
    args=["server.py"],
)

# 환경 변수 설정
params_with_env = StdioServerParameters(
    command="uv",
    args=["run", "server.py"],
    env={
        "DATABASE_URL": "postgresql://localhost/mydb",
        "API_KEY": "secret-key",
    },
)

# 작업 디렉토리 설정
params_with_cwd = StdioServerParameters(
    command="uv",
    args=["run", "server.py"],
    cwd="/path/to/project",
)

8.3 Streamable HTTP 클라이언트

Streamable HTTP 클라이언트는 원격에서 실행 중인 MCP 서버에 HTTP를 통해 연결한다. 서버가 이미 실행 중이어야 하며, 서버의 URL을 지정하여 연결한다.

기본 연결

import asyncio
from mcp.client.session import ClientSession
from mcp.client.streamable_http import streamable_http_client

async def main():
    # 원격 서버에 연결
    url = "http://localhost:8000/mcp"

    async with streamable_http_client(url) as (read, write, _):
        async with ClientSession(read, write) as session:
            await session.initialize()

            # 도구 목록 조회
            tools = await session.list_tools()
            for tool in tools.tools:
                print(f"  - {tool.name}: {tool.description}")

            # 도구 호출
            result = await session.call_tool("hello", {"name": "세계"})
            print(f"결과: {result.content[0].text}")

asyncio.run(main())
INFO

streamable_http_client()는 3개의 값을 반환한다. 세 번째 값은 세션 관련 객체인데, 대부분의 경우 사용하지 않으므로 _로 무시한다.

헤더 추가 (인증 등)

async def main():
    url = "https://api.example.com/mcp"

    # 인증 헤더 포함
    headers = {
        "Authorization": "Bearer your-token-here",
    }

    async with streamable_http_client(url, headers=headers) as (read, write, _):
        async with ClientSession(read, write) as session:
            await session.initialize()
            # ... 서버 기능 사용

8.4 리소스 읽기

클라이언트에서 서버가 제공하는 리소스를 조회하고 읽을 수 있다. list_resources()로 사용 가능한 리소스를 확인하고, read_resource()로 특정 리소스의 내용을 가져온다.

리소스 목록 조회 및 읽기

from pydantic import AnyUrl

async def read_resources(session: ClientSession):
    # 리소스 목록 조회
    resources = await session.list_resources()
    print("=== 사용 가능한 리소스 ===")
    for resource in resources.resources:
        print(f"  URI: {resource.uri}")
        print(f"  이름: {resource.name}")
        print(f"  설명: {resource.description}")
        print()

    # 특정 리소스 읽기
    content = await session.read_resource(AnyUrl("config://app"))
    print("=== config://app 리소스 내용 ===")
    for item in content.contents:
        print(item.text)

Resource Template 사용

async def read_user_profile(session: ClientSession, user_id: str):
    # Resource Template 목록 조회
    templates = await session.list_resource_templates()
    print("=== 리소스 템플릿 ===")
    for template in templates.resource_templates:
        print(f"  URI 템플릿: {template.uri_template}")
        print(f"  이름: {template.name}")
        print()

    # 템플릿에 값을 채워서 리소스 읽기
    uri = f"users://{user_id}/profile"
    content = await session.read_resource(AnyUrl(uri))
    for item in content.contents:
        print(f"사용자 {user_id} 프로필: {item.text}")
TIP

list_resources()는 정적 리소스(고정 URI)를, list_resource_templates()는 동적 리소스(URI 템플릿)를 반환한다. 둘 다 확인해야 서버가 제공하는 모든 리소스를 파악할 수 있다.

8.5 프롬프트 가져오기

서버가 제공하는 프롬프트 목록을 조회하고, 파라미터를 전달하여 완성된 프롬프트를 가져올 수 있다.

프롬프트 목록 조회

async def list_all_prompts(session: ClientSession):
    # 프롬프트 목록 조회
    prompts = await session.list_prompts()
    print("=== 사용 가능한 프롬프트 ===")
    for prompt in prompts.prompts:
        print(f"  이름: {prompt.name}")
        print(f"  설명: {prompt.description}")
        if prompt.arguments:
            print(f"  파라미터:")
            for arg in prompt.arguments:
                required = "필수" if arg.required else "선택"
                print(f"    - {arg.name} ({required}): {arg.description}")
        print()

프롬프트 가져오기

async def use_prompt(session: ClientSession):
    # 프롬프트 가져오기 (파라미터 전달)
    prompt_result = await session.get_prompt(
        "summarize",
        {"text": "MCP는 Model Context Protocol의 약자로...", "language": "ko"},
    )

    print("=== 생성된 프롬프트 메시지 ===")
    for message in prompt_result.messages:
        print(f"  역할: {message.role}")
        print(f"  내용: {message.content.text}")
        print()

8.6 실전 예제: MCP 클라이언트 CLI

서버에 연결하여 인터랙티브하게 도구를 선택하고 호출하는 CLI 클라이언트를 만들어 보자. 이 예제는 stdio 방식으로 서버에 연결하여 도구 목록을 표시하고, 사용자가 선택한 도구를 실행한다.

서버 코드 (server.py)

"""MCP 서버 — CLI 클라이언트와 함께 사용할 도구 모음"""
import json
import math
from datetime import datetime
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("demo-server")


@mcp.tool()
def add(a: float, b: float) -> str:
    """두 수를 더한다"""
    return str(a + b)


@mcp.tool()
def multiply(a: float, b: float) -> str:
    """두 수를 곱한다"""
    return str(a * b)


@mcp.tool()
def sqrt(number: float) -> str:
    """제곱근을 계산한다"""
    if number < 0:
        return "오류: 음수의 제곱근은 계산할 수 없다"
    return str(math.sqrt(number))


@mcp.tool()
def current_time() -> str:
    """현재 시각을 반환한다"""
    now = datetime.now()
    return now.strftime("%Y년 %m월 %d일 %H시 %M분 %S초")


@mcp.tool()
def word_count(text: str) -> str:
    """텍스트의 글자 수와 단어 수를 센다"""
    result = {
        "글자 수 (공백 포함)": len(text),
        "글자 수 (공백 제외)": len(text.replace(" ", "")),
        "단어 수": len(text.split()),
        "줄 수": len(text.splitlines()),
    }
    return json.dumps(result, ensure_ascii=False, indent=2)


@mcp.resource("server://info")
def server_info() -> str:
    """서버 정보를 반환한다"""
    return json.dumps({
        "name": "demo-server",
        "version": "1.0.0",
        "tools_count": 5,
    }, indent=2)


if __name__ == "__main__":
    mcp.run()

클라이언트 코드 (client.py)

"""MCP 클라이언트 CLI — 서버에 연결하여 도구를 인터랙티브하게 호출한다"""
import asyncio
import json
from mcp.client.session import ClientSession
from mcp.client.stdio import StdioServerParameters, stdio_client


async def display_server_info(session: ClientSession):
    """서버 정보를 표시한다"""
    print("\n" + "=" * 50)
    print("  MCP 클라이언트 CLI")
    print("=" * 50)

    # 리소스에서 서버 정보 읽기
    try:
        from pydantic import AnyUrl
        content = await session.read_resource(AnyUrl("server://info"))
        info = json.loads(content.contents[0].text)
        print(f"  서버: {info['name']} v{info['version']}")
    except Exception:
        print("  서버 정보를 가져올 수 없다")


async def display_tools(session: ClientSession) -> list:
    """도구 목록을 표시하고 반환한다"""
    tools_result = await session.list_tools()
    tools = tools_result.tools

    print("\n사용 가능한 도구:")
    print("-" * 40)
    for i, tool in enumerate(tools, 1):
        print(f"  {i}. {tool.name}")
        print(f"     {tool.description}")
        if tool.inputSchema.get("properties"):
            params = tool.inputSchema["properties"]
            param_strs = []
            for name, schema in params.items():
                param_type = schema.get("type", "any")
                param_strs.append(f"{name}: {param_type}")
            print(f"     파라미터: {', '.join(param_strs)}")
        print()

    return tools


def get_tool_arguments(tool) -> dict:
    """사용자로부터 도구 인자를 입력받는다"""
    properties = tool.inputSchema.get("properties", {})
    required = tool.inputSchema.get("required", [])
    args = {}

    if not properties:
        return args

    print(f"\n'{tool.name}' 도구의 파라미터를 입력한다:")
    for name, schema in properties.items():
        param_type = schema.get("type", "string")
        is_required = name in required
        marker = "(필수)" if is_required else "(선택, Enter로 건너뛰기)"

        value = input(f"  {name} [{param_type}] {marker}: ").strip()

        if not value and not is_required:
            continue

        # 타입 변환
        if param_type == "number":
            args[name] = float(value)
        elif param_type == "integer":
            args[name] = int(value)
        elif param_type == "boolean":
            args[name] = value.lower() in ("true", "1", "yes")
        else:
            args[name] = value

    return args


async def interactive_loop(session: ClientSession):
    """인터랙티브 루프 — 사용자와 대화하며 도구를 실행한다"""
    await display_server_info(session)
    tools = await display_tools(session)

    while True:
        print("-" * 40)
        choice = input(
            "도구 번호를 입력한다 (q: 종료, l: 목록 새로고침): "
        ).strip()

        if choice.lower() == "q":
            print("\n클라이언트를 종료한다.")
            break

        if choice.lower() == "l":
            tools = await display_tools(session)
            continue

        # 도구 선택
        try:
            tool_index = int(choice) - 1
            if tool_index < 0 or tool_index >= len(tools):
                print(f"  1~{len(tools)} 사이의 번호를 입력한다.")
                continue
        except ValueError:
            print("  올바른 번호를 입력한다.")
            continue

        selected_tool = tools[tool_index]
        print(f"\n선택한 도구: {selected_tool.name}")

        # 인자 입력
        try:
            args = get_tool_arguments(selected_tool)
        except (ValueError, KeyboardInterrupt):
            print("\n  입력이 취소되었다.")
            continue

        # 도구 호출
        print(f"\n'{selected_tool.name}' 도구를 호출한다...")
        try:
            result = await session.call_tool(selected_tool.name, args)
            print("\n결과:")
            print("-" * 40)
            for content_item in result.content:
                print(content_item.text)
            print("-" * 40)
        except Exception as e:
            print(f"\n오류 발생: {e}")


async def main():
    """메인 함수 — 서버에 연결하고 인터랙티브 루프를 실행한다"""
    server_params = StdioServerParameters(
        command="uv",
        args=["run", "server.py"],
    )

    print("서버에 연결 중...")
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            print("서버에 연결되었다!")

            await interactive_loop(session)


if __name__ == "__main__":
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        print("\n\n프로그램이 종료되었다.")

이 CLI 클라이언트는 학습용 예제이다. 프로덕션 환경에서는 click이나 typer 같은 CLI 프레임워크를 사용하면 더 풍부한 인터페이스를 구현할 수 있다. 또한 LLM과 연동하면 자연어로 도구를 선택하고 호출하는 에이전트를 만들 수 있다.

INFO
